# Lanobot 构建计划

## 项目概述

**目标**：使用 LangChain 1.2 完全重构 nanobot（超轻量级个人AI助手），创建 lanobot 新架构。

**核心约束**：
- 使用 `uv` 管理 Python 包
- 目标平台：Windows 11
- 禁止使用废弃 API（如 `create_react_agent`、`prompt=`、`MemorySaver()`）
- 必须使用新 API（`create_agent`、`system_prompt`、`InMemorySaver`）
- **LangChain 版本**：1.2.0 ≤ version < 2.0

---

**首选**：DeepSeek + SiliconFlow（通过 OpenAI 兼容 API）

```python
from langchain_openai import ChatOpenAI

def create_llm(provider: str, model: str, api_key: str = None, api_base: str = None):
    """创建LangChain LLM实例"""

    # DeepSeek

            model=model or "deepseek-chat",
            api_key=api_key,
            base_url=api_base or "https://api.deepseek.com/v1"

            base_url=api_base
```

```python
# agent/prompt.py
from pathlib import Path

def load_system_prompt(templates_dir: Path) -> str:
    """加载系统提示词（按优先级拼接）"""
    template_files = [
        "SOUL.md",       # 助手身份
        "AGENTS.md",     # 行为规范
        "TOOLS.md",      # 工具约束
        "USER.md",       # 用户信息
    ]

    prompts = []
    for filename in template_files:
        file_path = templates_dir / filename
        if file_path.exists():
            content = file_path.read_text(encoding="utf-8")
            prompts.append(f"<!-- {filename} -->\n{content}")

    return "\n\n".join(prompts)
```

### 主入口设计 (main.py)

```python
# main.py
import asyncio
import logging
from pathlib import Path

from config.loader import load_config
from agent.graph import AgentGraph
from bus.queue import MessageBus
from providers.langchain_wrapper import create_llm
from channels.manager import ChannelManager
from memory.checkpointer import get_checkpointer
from agent.prompt import load_system_prompt
from tools.registry import get_tools


async def main():
    """Lanobot 主入口"""

    # 1. 加载配置
    config = load_config("config.json")
    setup_logging(config.log_level)

    logger = logging.getLogger("lanobot")
    logger.info("Lanobot 启动中...")

    # 2. 加载系统提示词
    templates_dir = Path(config.templates_dir)
    system_prompt = load_system_prompt(templates_dir)
    logger.info(f"已加载系统提示词 from {templates_dir}")

    # 3. 创建 LLM
    llm = create_llm(
        provider=config.provider,
        model=config.model,
        api_key=config.api_key,
        api_base=config.api_base
    )
    logger.info(f"已创建 LLM: {config.provider}/{config.model}")

    # 4. 创建 Checkpointer（短期记忆）
    checkpointer = get_checkpointer()
    logger.info("已创建 Checkpointer")

    # 5. 获取工具列表
    tools = get_tools(config)
    logger.info(f"已加载 {len(tools)} 个工具")

    # 6. 创建 Agent
    agent = AgentGraph(
        model=llm,
        tools=tools,
        checkpointer=checkpointer,
        system_prompt=system_prompt
    )
    logger.info("已创建 Agent")

    # 7. 创建消息总线
    bus = MessageBus()
    logger.info("已创建消息总线")

    # 8. 启动渠道
    channel_manager = ChannelManager(config.channels, agent, bus)
    await channel_manager.start_all()
    logger.info(f"已启动渠道: {config.channels}")

    # 9. 消息处理循环
    logger.info("Lanobot 启动完成，开始处理消息...")
    try:
        while True:
            message = await bus.consume_inbound()
            logger.debug(f"收到消息: {message.content[:50]}...")
            response = await agent.invoke(message.content, message.thread_id)
            await bus.publish_outbound(response)
    except KeyboardInterrupt:
        logger.info("收到退出信号")
    finally:
        await channel_manager.stop_all()
        logger.info("Lanobot 已停止")


def setup_logging(level: str):
    """配置日志系统"""
    logging.basicConfig(
        level=getattr(logging, level.upper()),
        format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
        handlers=[
            logging.StreamHandler(),
            logging.FileHandler("workspace/logs/lanobot.log", encoding="utf-8")
        ]
    )


if __name__ == "__main__":
    asyncio.run(main())
```

### 日志系统设计

```python
# utils/logging.py
import logging
import logging.handlers
from pathlib import Path
from typing import Optional

class LanobotLogger:
    """Lanobot 日志系统"""

    def __init__(self, log_dir: Path, log_level: str = "INFO"):
        self.log_dir = log_dir
        self.log_level = log_level
        self._setup_logging()

    def _setup_logging(self):
        """配置日志系统"""
        # 确保日志目录存在
        self.log_dir.mkdir(parents=True, exist_ok=True)

        # 创建 logger
        logger = logging.getLogger("lanobot")
        logger.setLevel(getattr(logging, self.log_level.upper()))

        # 控制台输出
        console_handler = logging.StreamHandler()
        console_handler.setLevel(logging.INFO)
        console_format = logging.Formatter(
            "%(levelname)s - %(message)s"
        )
        console_handler.setFormatter(console_format)

        # 文件输出（带轮转）
        file_handler = logging.handlers.RotatingFileHandler(
            self.log_dir / "lanobot.log",
            maxBytes=10 * 1024 * 1024,  # 10MB
            backupCount=5,
            encoding="utf-8"
        )
        file_handler.setLevel(logging.DEBUG)
        file_format = logging.Formatter(
            "%(asctime)s - %(name)s - %(levelname)s - %(filename)s:%(lineno)d - %(message)s"
        )
        file_handler.setFormatter(file_format)

        # 添加处理器
        logger.addHandler(console_handler)
        logger.addHandler(file_handler)

    def get_logger(self, name: str) -> logging.Logger:
        """获取子模块 logger"""
        return logging.getLogger(f"lanobot.{name}")
```

**核心功能**：
- 会话持久化：基于 `workspace/sessions/` 目录的 **JSONL 文件**存储（追加写入）
- 多渠道隔离：通过 `channel:chat_id` 格式的 **key** 隔离不同渠道
- 过期清理：默认 30 天过期，非 pinned 会话自动清理
- **LLM 摘要压缩**：10万 tokens 触发 LLM 生成摘要，保存到 `sessions/summaries/` 目录
- **异步标题生成**：调用 LLM 自动生成会话标题
- **HistoryManager 集成**：`load_to_history_manager()` 方法加载会话到内存历史
- /pin, /unpin, /title 命令支持

**SessionMetadata 新增字段**：
- `key`: channel:chat_id
- `title_source`: "auto" | "user"
- `updated_at`: 更新时间
- `last_summarized_at`: 上次摘要时间
- `summary_token_count`: 摘要 token 数

**数据目录结构**：
```

    └── dingtalk_88888888.summary.jsonl
```

```python
# session/ 使用示例
from session import SessionManager, create_session_manager
```
